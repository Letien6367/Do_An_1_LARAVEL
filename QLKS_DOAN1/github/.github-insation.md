
## Database
CREATE TABLE user
(
	id INT IDENTITY(1,1) PRIMARY KEY,
	name NVARCHAR NOT NULL,
	email NVARCHAR NOT NULL,
	SoDienThoai NVARCHAR NOT NULL,
	password NVACHAR NOT NULL,
	VAITRO NVARCHAR NOT NULL
);
CREATE TABLE LOAIPHONG
(
	MaLoaiPhong INT IDENTITY(1,1) PRIMARY KEY,
	TenLoaiPhong NVARCHAR NOT NULL,

);
CREATE TABLE TRANGTHAIPHONG
(
	MaTrangThai INT IDENTITY(1,1) PRIMARY KEY,
	TenTrangThai NVARCHAR NOT NULL,
	
);
CREATE TABLE PHONG 
(
	MaPhong INT IDENTITY(1,1) PRIMARY KEY ,
	TenPhong NVARCHAR NOT NULL,
	SoNguoi INT NOT NULL,
	GiaPhong DECIMAL(12,2),
	MaTrangThai INT NOT NULL,
	MaLoaiPhong INT NOT NULL,
	CONSTRAINT FK_PHONG_LoaiPhong FOREIGN KEY (MaLoaiPhong) REFERENCES LoaiPhong(MaLoaiPhong),
	CONSTRAINT FK_PHONG_TrangThai FOREIGN KEY (MaTrangThai) REFERENCES TRANGTHAIPHONG(MaTrangThai)

);
CREATE TABLE KHACHHANG
(
	MaKhachHang INT IDENTITY(1,1) PRIMARY KEY,
	MaTaiKhoan INT,
	TenKhachHang NVARCHAR NOT NULL,
	NgaySinh NVARCHAR NOT NULL,
	SoDienThoai NVARCHAR NOT NULL,
	DiaChi NVARCHAR NOT NULL,
	GiayChungMinh NVARCHAR NOT NULL,
	CONSTRAINT FK_KHACHHANG_TAIKHOAN FOREIGN KEY (MaTaiKhoan) REFERENCES user(MaTaiKhoan)
);
CREATE TABLE TRANGTHAIDATPHONG
(
	MaTrangThaiDP INT IDENTITY(1,1) PRIMARY KEY,
	TenTrangThaiDP NVARCHAR NOT NULL,
);
CREATE TABLE DATPHONG
(
	MaDatPhong INT IDENTITY(1,1) PRIMARY KEY,
	MaPhong INT,
	MaKhachHang INT,
	NgayDatPhong DATE NOT NULL,
	NgayTraPhong DATE NOT NULL,
	MaTrangThaiDP INT,
	CONSTRAINT FK_DATPHONG_PHONG FOREIGN KEY (MaPhong) REFERENCES PHONG(MaPhong),
	CONSTRAINT FK_DATPHONG_KHACHANG FOREIGN KEY (MaKhachHang) REFERENCES KHACHHANG(MaKhachHang),
	CONSTRAINT FK_DATPHONG_TRANGTHAIDATPHONG FOREIGN KEY (MaTrangThaiDP) REFERENCES TRANGTHAIDATPHONG(MaTrangThaiDP)
);
CREATE TABLE HOADON
(
	MaHoaDon INT IDENTITY(1,1) PRIMARY KEY,
	MaPhong INT,
	MaDatPhong INT,
	NgayLapHD DATE NOT NULL,
	TongTien DECIMAL(12,2),
	CONSTRAINT FK_HOADON_PHONG FOREIGN KEY (MaPhong) REFERENCES PHONG(MaPhong),
	CONSTRAINT FK_HOADON_DATPHONG FOREIGN KEY (MaDatPhong) REFERENCES DATPHONG(MaDatPhong)
);
